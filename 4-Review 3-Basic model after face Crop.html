model1 = Sequential()

model1.add(MaxPooling2D(pool_size = (2,2)))

model1.add(MaxPooling2D(pool_size = (2,2)))

model1.add(Flatten())
model1.add(Dense(64, activation = 'relu'))

model1.compile(optimizer ='adam', loss = 'categorical_crossentropy', metrics = ['accuracy'])
